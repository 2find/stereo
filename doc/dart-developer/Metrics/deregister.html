void deregister (Metric metric)

Deregister Metrics to make them not visible to Observatory.

Implementation 

static void deregister(Metric metric) {
  if (metric is! Metric) {
    throw new ArgumentError('metric must be a Metric');
  }
  _metrics.remove(metric.name);
}
