void finish ()

Finish the last synchronous operation that was started.

Implementation 

void finish() {
  if (_isProduct) {
    return;
  }
  if (_stack.length == 0) {
    throw new StateError('Uneven calls to start and finish');
  }
  // Pop top item off of stack.
  var block = _stack.removeLast();
  block._finish();
}
