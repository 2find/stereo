bool containsRectangle (Rectangle<num> another)

Tests whether this entirely contains another.

Implementation 

bool containsRectangle(Rectangle<num> another) {
  return left <= another.left &&
      left + width >= another.left + another.width &&
      top <= another.top &&
      top + height >= another.top + another.height;
}
